# Git Merge Changes Branch into Main and Push to GitHub

# File Name Suggestion:

# git_merge_changes_to_main_and_push.txt

# -------------------------------------------------------

# STEP 1 - Verify Current Branch

# Purpose:

# Check current branch.

# -------------------------------------------------------

git branch

# -------------------------------------------------------

# STEP 2 - Check Working Tree

# Purpose:

# Verify pending changes.

# -------------------------------------------------------

git status

# -------------------------------------------------------

# STEP 3 - Save Current Work

# Purpose:

# Commit uncommitted changes if any.

# Skip if clean.

# -------------------------------------------------------

git add .
git commit -m "Save current work before merge"

# -------------------------------------------------------

# STEP 4 - Download Latest Remote Changes

# Purpose:

# Sync branch information.

# -------------------------------------------------------

git fetch --all

# -------------------------------------------------------

# STEP 5 - Switch to Main Branch

# Purpose:

# Move to target branch.

# -------------------------------------------------------

git checkout main

# -------------------------------------------------------

# STEP 6 - Update Local Main

# Purpose:

# Get latest GitHub main changes.

# -------------------------------------------------------

git pull origin main

# -------------------------------------------------------

# STEP 7 - Merge Changes into Main

# Purpose:

# Bring all changes branch commits into main.

# -------------------------------------------------------

git merge changes

# -------------------------------------------------------

# STEP 8 - Resolve Conflicts If Any

# Purpose:

# Fix conflicts manually.

# Then stage files.

# -------------------------------------------------------

git add .

# -------------------------------------------------------

# STEP 9 - Complete Merge Commit

# Purpose:

# Finish merge if conflicts occurred.

# Skip if auto merged.

# -------------------------------------------------------

git commit -m "Merge changes into main"

# -------------------------------------------------------

# STEP 10 - Push Main to GitHub

# Purpose:

# Upload merged main branch.

# -------------------------------------------------------

git push origin main

# -------------------------------------------------------

# STEP 11 - Verify Status

# Purpose:

# Confirm clean repository.

# -------------------------------------------------------

git status

# -------------------------------------------------------

# STEP 12 - Verify History

# Purpose:

# Confirm merge commit exists.

# -------------------------------------------------------

git log --oneline --graph --all --decorate

# -------------------------------------------------------

# STEP 13 - Verify Remote

# Purpose:

# Confirm correct GitHub repository.

# -------------------------------------------------------

git remote -v
